\section{Use-case và kịch bản vận hành (Usecase scenario)}

Dưới đây là 8 kịch bản vận hành trọng tâm của hệ thống website doanh nghiệp, được mô tả chi tiết các luồng xử lý và ràng buộc dữ liệu.

% ========================= UC-ADM-001: LOGIN =========================
\begin{table}[H]
\centering
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{|p{3.5cm}|p{9.5cm}|}
\hline
\textbf{Use-case ID} & UC-ADM-001 \\ \hline
\textbf{Use-case} & Đăng nhập hệ thống quản trị (Common Task) \\ \hline
\textbf{Actor} & Quản trị viên \\ \hline
\textbf{Description} & Cho phép Quản trị viên đăng nhập vào hệ thống quản lý nội dung (CMS) để điều phối hoạt động website. \\ \hline
\textbf{Precondition} & Quản trị viên đã có tài khoản hợp lệ trong cơ sở dữ liệu. \\ \hline
\textbf{Postcondition} & Phiên làm việc (Session) được khởi tạo; Quản trị viên truy cập vào Dashboard. \\ \hline
\textbf{Trigger} & Quản trị viên truy cập trang \texttt{/admin} và nhập thông tin. \\ \hline
\textbf{Normal Flow} & 
1. Hệ thống hiển thị form đăng nhập. \newline
2. Quản trị viên nhập Tên đăng nhập và Mật khẩu. \newline
3. Hệ thống kiểm tra thông tin trong cơ sở dữ liệu MySQL. \newline
4. Hệ thống xác thực mật khẩu (sử dụng \texttt{password\_verify}). \newline
5. Nếu hợp lệ, hệ thống tạo phiên đăng nhập và chuyển hướng vào Dashboard. \\ \hline
\textbf{Alternative Flow} & Không có. \\ \hline
\textbf{Exception Flow} & 
E1 - Sai thông tin: Hệ thống báo lỗi "Tài khoản hoặc mật khẩu không chính xác" và yêu cầu nhập lại. \newline
E2 - Lỗi kết nối DB: Hệ thống hiển thị thông báo lỗi kỹ thuật. \\ \hline
\end{tabular}
\end{table}

\hline
\textbf{Use-case ID} & UC-ADM-002 \\ \hline
\textbf{Use-case} & Cấu hình thông tin liên hệ Website (Task 1) \\ \hline
\textbf{Actor} & Quản trị viên \\ \hline
\textbf{Description} & Cập nhật các thông tin cơ bản như Hotline, Email, Địa chỉ công ty hiển thị tại Header/Footer. \\ \hline
\textbf{Precondition} & Quản trị viên đã đăng nhập thành công. \\ \hline
\textbf{Postcondition} & Thông tin liên hệ mới được cập nhật trên toàn bộ website. \\ \hline
\textbf{Trigger} & Quản trị viên truy cập mục "Cài đặt chung" từ Dashboard. \\ \hline
\textbf{Normal Flow} & 
1. Hệ thống hiển thị các thông tin cấu hình hiện tại từ bảng \texttt{settings}. \newline
2. Quản trị viên thay đổi các trường dữ liệu cần thiết. \newline
3. Quản trị viên nhấn nút "Lưu thay đổi". \newline
4. Hệ thống cập nhật dữ liệu vào MySQL. \newline
5. Hệ thống xóa cache (nếu có) và hiển thị thông báo thành công. \\ \hline
\textbf{Alternative Flow} & A1 - Hủy chỉnh sửa: Quản trị viên nhấn "Hủy", hệ thống quay về dữ liệu cũ. \\ \hline
\textbf{Exception Flow} & E1 - Dữ liệu không hợp lệ: Số điện thoại hoặc Email sai định dạng, hệ thống yêu cầu kiểm tra lại. \\ \hline
\end{tabular}
\end{table}

\textbf{Description} & Xem, phân loại và cập nhật trạng thái các yêu cầu liên hệ gửi từ khách hàng. \\ \hline
\textbf{Precondition} & Đã có dữ liệu khách hàng gửi từ form liên hệ. \\ \hline
\textbf{Postcondition} & Trạng thái yêu cầu được cập nhật, ghi chú xử lý được lưu lại. \\ \hline
\textbf{Trigger} & Quản trị viên mở mục "Hộp thư liên hệ". \\ \hline
\textbf{Normal Flow} & 
1. Hệ thống liệt kê danh sách liên hệ theo thời gian mới nhất. \newline
2. Quản trị viên nhấn xem chi tiết một yêu cầu. \newline
3. Quản trị viên thực hiện liên hệ tư vấn cho khách hàng. \newline
4. Quản trị viên cập nhật trạng thái thành "Đã xử lý" và thêm ghi chú. \newline
5. Hệ thống lưu trạng thái mới. \\ \hline
\textbf{Alternative Flow} & A1 - Xóa liên hệ: Quản trị viên xóa các tin nhắn spam hoặc tin nhắn rác. \\ \hline
\textbf{Exception Flow} & E1 - Lỗi cập nhật: Hệ thống báo lỗi khi không thể lưu trạng thái mới. \\ \hline
\end{tabular}
\end{table}
